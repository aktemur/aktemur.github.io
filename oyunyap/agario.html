float x = 200;
float y = 300;
float yaricap = 20;
float hizX = 0;
float hizY = 0;

float yemX = 40;
float yemY = 40;
float yemYaricap = 10;

void setup() {
  size(400, 600);
}

void draw() {
  background(150, 150, 150);
  ellipse(x, y, yaricap*2, yaricap*2);
  ellipse(yemX, yemY, yemYaricap*2, yemYaricap*2);

  if (dist(yemX, yemY, x, y) < yaricap + yemYaricap) {
    yaricap = yaricap + 2;
    yemX = random(0, width);
    yemY = random(0, height);
  }

  hizX = (mouseX - x) / yaricap;
  hizY = (mouseY - y) / yaricap;

  x = x + hizX;
  y = y + hizY;
}

void keyPressed() {
  if (keyCode == LEFT) {
    hizX = -3;
  }
  if (keyCode == RIGHT) {
    hizX = 3;
  }
  if (keyCode == UP) {
    hizY = -3;
  }
  if (keyCode == DOWN) {
    hizY = 3;
  }
}

void keyReleased() {
  if (keyCode == LEFT) {
    hizX = 0;
  }
  if (keyCode == RIGHT) {
    hizX = 0;
  }
  if (keyCode == UP) {
    hizY = 0;
  }
  if (keyCode == DOWN) {
    hizY = 0;
  }
}
