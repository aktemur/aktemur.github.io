void setup() {
  size(400, 400);
  fill(0, 0, 255);
}

int aci = 0;

void draw() {
  // background(255, 255, 255);
  text("sin " + aci, 0, 10);
  float radianAci = radians(aci*5);
  float sinus = sin(radianAci)*50 + 100;
  float kosinus = cos(radianAci)*50 + 100;
  text(" = " + sinus, 60, 10);

  fill(0, 0, 255);
  ellipse(aci, sinus, 5, 5);
  fill(255, 0, 0);
  ellipse(aci, kosinus, 5, 5);

  aci = aci + 1;
}
