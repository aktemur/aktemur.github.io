void setup() {
  size(400, 400);
  fill(0, 0, 255);
}

int merkezX = 200;
int merkezY = 200;
int aci = 0;
float d = 100;

void draw() {
  background(255, 255, 255);

  float alfa = radians(aci);

  float x = d*cos(alfa) + merkezX;
  float y = d*sin(alfa) + merkezY;

  ellipse(x, y, 5, 5);

  ellipse(merkezX, merkezY, 5, 5);
  line(merkezX, merkezY, x, y);

  text("Alfa: " + aci, 5, height-10);
  aci = aci + 1;
}
